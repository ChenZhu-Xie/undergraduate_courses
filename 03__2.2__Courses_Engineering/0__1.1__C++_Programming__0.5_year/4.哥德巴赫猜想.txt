#include<stdio.h>
#include<math.h>
void main()
{
	int n,i,i1,j1,i2,j2;
	printf("请输入一个正整数n,输入完毕后按enter键实现此程序功能:");
	scanf("%d",&n);
	for(i=4;i<=n;i=i+2)
	{
		for(i1=2;i1<=i/2;i1++)	/*根据加法交换律,我们只需要找出并利用数学上[i/2]以内的所有素数*/
		{
			for(j1=2;j1<=sqrt(i1);j1++)	/*根据乘法交换律,我们只需尝试≤根号i1的所有可能的i1的因子*/
				if(i1%j1==0)	break;
				if(j1>sqrt(i1))	/*此时若i1是≥4的质数,则进入大括号,否则滤过大括号往下;若i1为2、3,则也会直接来到这里并往下*/
				{
					i2=i-i1;
					for(j2=2;j2<=sqrt(i2);j2++)
						if(i2%j2==0)	break;
						if(j2>sqrt(i2))	/*此时i2也是质数,i2一般不存在=2、3的情况,即此时i2也是≥4的质数;除非i太小，此时i-i1得到的i2可能为2或3,不过仍可通过判断;若i2非质,则什么都不做,跳出大括号,*/
						{/*这个大括号和里面的[break;]若都同时去掉，那么表示第二层的for函数会执行所有的i1=[2,i/2],即所有符合条件的(质数+质数);这里表示找到一个即可,仅为了说明其存在性,*/
							printf("%d=%d+%d\n",i,i1,i2);
							break;
						}
				}
		}
	}
}
