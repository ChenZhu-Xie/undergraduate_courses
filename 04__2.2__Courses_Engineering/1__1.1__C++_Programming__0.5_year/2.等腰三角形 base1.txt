#include<stdio.h>
void main()
{
	int x,y,i,j1,j2;          /*注意:(x,y)是计算机坐标系下的,(i,j)是行列式坐标系下的所以(i,j)对应(y,x)*/
	printf("请按照以下格式输入您所想绘画的等腰三角形图案的上顶点相对于按了enter后的光标处的坐标:x,y(计算机坐标系下的,x≤39)\n");
	scanf("%d,%d",&x,&y);
	for(i=1;i<=y+x+1;i++)   /*控制行数*/
	{
		if(i<=y)
		printf("\n");
		else
		{for(j1=1;j1<=x+(y+1)-i;j1++)
			printf(" ");
			for(j2=1;j2<=2*(i-y)-1;j2++)
			printf("*");
			printf("\n");
		}
	}
}
