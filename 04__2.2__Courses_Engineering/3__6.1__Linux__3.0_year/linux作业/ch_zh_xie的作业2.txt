作业：写一个脚本，接收一个参数，如果不给参数，提示用法；如果参数指定的文件不存在，则显示不存在，并退出(单分支if语句)；否则，如果是一个普通文件，显示之；如果是一个目录，也显示之；否则如果又不是文件又不是目录，则显示无法识别该文件(多分支if语句)。

#!/bin/bash
if [ $# -ne 1 ];then
  echo  'Usage: ./exist.sh Arg1'
  exit 3
fi
if [ ! -e $1 ];then
  echo  "No such a file."
  exit 4
else
  if [ -f $1 ];then
    echo 'This is a file:'
    cat -n $1
    exit 5
  elif [ -d $1 ];then
    echo 'This is a directory:'
    cat -n $1
    exit 6
  else
    echo 'Cannot recognize this file!'
    exit 7
  fi
fi
